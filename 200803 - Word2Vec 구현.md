class Matmul:
  def __init__(self,W):
    self.params=[W]
    self.grads=[np.zeros_like(W)]
    self.x=None
    
  def forward(self,x):
    W,=self.params
    out=np.matmul(x,W)
    self.x=x
    return out
    
  def backward(self,dout):
    W,=self.params
    dx=np.matmul(dout,W.T)
    dW=np.matmul(self.x.T,dout)
    self.grads[0][...]=dW
    return dx
```
